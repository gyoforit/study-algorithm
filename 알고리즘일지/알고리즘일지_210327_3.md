def zoo(N, data):
    if max(data) >= N:
        return 0
    else:
        X = max(data) + 1
        Y = N-X
        for i in range(0, X):
            if Y != 0:
                if i <= Y - 1:
                    if data.count(i) != 2:
                        return 0
                else:
                    if data.count(i) != 1:
                        return 0
            else:
                if data.count(i) != 1:
                    return 0
        if X != Y:
            return (2**Y)*2
        else:
            return 2**Y

N = int(input())
data = list(map(int, input().split()))
print(zoo(N, data))
