def formalize(s):
    s = s.strip().replace(',', ';').lower()
    for c in "{[": s = s.replace(c, '(')
    for c in "}]": s = s.replace(c, ')')
    while '  ' in s: s = s.replace('  ', ' ')
    for c in "().,;:": s = s.replace(' '+c, c).replace(c+' ',c)
    return s

for TEST in range(int(input())):
    a = formalize(input())
    b = formalize(input())
    print(f"Data Set {TEST+1}: {'not ' if a != b else ''}equal")
    print()
