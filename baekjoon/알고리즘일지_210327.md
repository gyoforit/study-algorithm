```python
# 재귀로 풀면 시간초과
def DFS(idx, S, total):
    global mn
    if total >= mn:
        return

    if idx == N:
        if total < mn:
            mn = total
            return

    for i in range(N):
        if check[i] == 0:
            NS = S+pp[i]
            Ntotal = total+NS
            check[i] = 1
            DFS(idx+1, NS, Ntotal)
            check[i] = 0

N = int(input())
pp = list(map(int, input().split()))
check = [0]*N
mn = 987654321
DFS(0, 0, 0)
print(mn)
```
